void loop() {
 String minta = "";

  while(Serial.available()>0) {
    minta += char (Serial.read()); 
  }
  minta.trim();

  if (minta == "Ya") {

    kirimdata();
  }
  minta = "";
  delay(1000);
}
 
void kirimdata() {
  for(int i=0;i<10;i++) { 
  buf[i]=analogRead(analogInPin);
  delay(10);
 }
 for(int i=0;i<9;i++) {
  for(int j=i+1;j<10;j++){
   if(buf[i]>buf[j]){
    temp=buf[i];
    buf[i]=buf[j];
    buf[j]=temp;
   }
  }
 }
 avgValue=0;
 for(int i=2;i<8;i++)
 avgValue+=buf[i];
 
 float pHVol=(float)avgValue*5.0/1024/4;
 float phValue = 5.70 * pHVol -15.5 ;
 phValue=14.2 - phValue;

 if(phValue <=8) {
  digitalWrite(ledPinHijau, HIGH);
  digitalWrite(ledPinMerah, LOW);
 }
 else if(phValue >8) {
  digitalWrite(ledPinHijau, LOW);
  digitalWrite(ledPinMerah, HIGH);
 }
  if(phValue <=6) {
  digitalWrite(ledPinHijau, LOW);
  digitalWrite(ledPinMerah, HIGH);
 }

///////////////////////////////////////////////////////////////////////////
//program Turbidity
  int sensorValue = analogRead(sensorPin);
  int turbidity = map(sensorValue, 0,720, 100, 0);
  delay(100);
  if (turbidity < 10) {
    digitalWrite(7, HIGH);
    digitalWrite(8, LOW);
    digitalWrite(9, LOW);
  }
  if ((turbidity > 20) && (turbidity < 50)) {
    digitalWrite(7, LOW);
    digitalWrite(8, HIGH);
    digitalWrite(9, LOW);
  }
  if (turbidity > 50) {
    digitalWrite(7, LOW);
    digitalWrite(8, LOW);
    digitalWrite(9, HIGH);
  }
 
  lcd.clear();
  lcd.setCursor(1,0);
  lcd.print("Deteksi Limbah Cair");
  lcd.setCursor(0, 2);
           //012345678901234567890
  lcd.print("pHValue = ");
  lcd.setCursor(10, 2);
  lcd.print(phValue, 2);
  lcd.setCursor(0, 3);
           //012345678901234567890
  lcd.print("    TBD = ");
  lcd.setCursor(10, 3);
  lcd.print(turbidity);
  lcd.print("%");
  delay(50);
  
  Serial.print(phValue);
  Serial.print("#");
  Serial.println(turbidity);
  delay(500);
  //String datakirim = String(phValue) + "#" + String(turbidity);
  //Serial.println(datakirim);
}
